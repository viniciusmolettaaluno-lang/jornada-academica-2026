🎯 Objetivo do Trabalho

Realizar uma análise comparativa avançada entre diferentes LLMs (ChatGPT, Claude, Grok) na resolução de problemas computacionais com estruturas dinâmicas em Python. O trabalho envolve a implementação de soluções para análise de treliças planas isostáticas e um estudo crítico detalhado do desempenho de cada modelo.

  • Implementação do problema de treliças planas por três LLMs distintos
  • Análise de forças internas (tração/compressão) e reações de apoio
  • Geração de gráficos: treliça original, mapa de calor, barras e histograma
  • Comparação detalhada entre os códigos gerados

# Código gerado pelo ChatGPT
# Função de eliminação de Gauss manual

def gauss(A, b):
    n = len(b)
    # Matriz aumentada
    M = []
    for i in range(n):
        linha = A[i][:] + [b[i]]
        M.append(linha)
    
    # Eliminação
    for k in range(n):
        # Pivotamento parcial
        maior = abs(M[k][k])
        linha_pivo = k
        for i in range(k + 1, n):
            if abs(M[i][k]) > maior:
                maior = abs(M[i][k])
                linha_pivo = i
        M[k], M[linha_pivo] = M[linha_pivo], M[k]
        
        # Verificação de singularidade
        if abs(M[k][k]) < 1e-12:
            raise ValueError("Sistema singular ou mecanismo estrutural.")
        
        # Eliminação
        for i in range(k + 1, n):
            fator = M[i][k] / M[k][k]
            for j in range(k, n + 1):
                M[i][j] -= fator * M[k][j]
    
    # Retrossubstituição
    x = [0.0] * n
    for i in range(n - 1, -1, -1):
        soma = 0.0
        for j in range(i + 1, n):
            soma += M[i][j] * x[j]
        x[i] = (M[i][n] - soma) / M[i][i]
    
    return x

# Código gerado pelo Claude IA
# Classe principal da treliça

class TrelicaPlana:
    def __init__(self, nos, barras, cargas, apoios):
        """
        nos: [(x1,y1), (x2,y2), ...]
        barras: [(nó_i, nó_j), ...]
        cargas: [(Fx,Fy), ...]
        apoios: "fixo" -> reage em X e Y, "movel" -> reage em Y, "livre" -> sem reação
        """
        self.nos = nos
        self.barras = barras
        self.cargas = cargas
        self.apoios = apoios
        self.N = len(nos)
        self.B = len(barras)
        self.definir_incognitas()
    
    def montar_sistema(self):
        """Monta a matriz de equilíbrio do sistema"""
        num_incognitas = len(self.nom_variaveis)
        A = np.zeros((2 * self.N, num_incognitas))
        b = np.zeros(2 * self.N)
        
        # Contribuição das barras
        for idx_barra, (i, j) in enumerate(self.barras):
            xi, yi = self.nos[i]
            xj, yj = self.nos[j]
            dx = xj - xi
            dy = yj - yi
            L = math.sqrt(dx**2 + dy**2)
            c = dx / L
            s = dy / L
            A[2 * i][idx_barra] += c
            A[2 * i + 1][idx_barra] += s
            A[2 * j][idx_barra] -= c
            A[2 * j + 1][idx_barra] -= s
        
        return A, b

# Código gerado pelo Grok
# Implementação manual de Gauss-Jordan

def gauss_jordan(A, b, tol=1e-10):
    """
    Resolve Ax = b pelo método de Gauss-Jordan (implementação manual).
    Retorna solução x e indica se o sistema é singular.
    """
    n = len(b)
    # Matriz aumentada
    M = np.hstack((A.astype(float), b.reshape(-1, 1)))
    
    # Eliminação para frente
    for i in range(n):
        # Pivotamento parcial
        max_row = np.argmax(np.abs(M[i:, i])) + i
        if abs(M[max_row, i]) < tol:
            return None, True  # Singular
        # Troca de linhas
        M[[i, max_row]] = M[[max_row, i]]
        # Normaliza pivô
        pivot = M[i, i]
        M[i] = M[i] / pivot
        # Elimina coluna
        for k in range(n):
            if k != i:
                factor = M[k, i]
                M[k] = M[k] - factor * M[i]
    
    x = M[:, -1]
    return x, False
