💻 Código Exemplo (Abordagem 1 - Mapa de Navegação Vetorial)

# Abordagem 1 — Mapa de Navegação Vetorial
# Visualização com Matplotlib

import math
import matplotlib.pyplot as plt

# Entrada
print("SIMULADOR DE NAVEGAÇÃO VETORIAL\n")
x = float(input("Deslocamento no eixo X (Leste/Oeste): "))
y = float(input("Deslocamento no eixo Y (Norte/Sul): "))

# Cálculos
modulo = math.sqrt(x**2 + y**2)
ux = x / modulo
uy = y / modulo
angulo = math.degrees(math.atan2(y, x))

# Direção cardinal
if x > 0:
    horizontal = "Leste"
elif x < 0:
    horizontal = "Oeste"
else:
    horizontal = ""

if y > 0:
    vertical = "Norte"
elif y < 0:
    vertical = "Sul"
else:
    vertical = ""

direcao = f"{vertical} {horizontal}".strip()

# Resultados
print("\n===== RESULTADOS ====")
print(f"Vetor: ({x}, {y})")
print(f"Módulo: {modulo:.2f}")
print(f"Vetor unitário: ({ux:.2f}, {uy:.2f})")
print(f"Ângulo: {angulo:.2f}°")
print(f"Direção: {direcao}")

# Visualização
plt.figure(figsize=(7, 7))
plt.quiver(0, 0, x, y, angles='xy', scale_units='xy', scale=1, color='blue')
plt.quiver(0, 0, ux, uy, angles='xy', scale_units='xy', scale=1, color='red')
limite = max(abs(x), abs(y)) + 2
plt.xlim(-limite, limite)
plt.ylim(-limite, limite)
plt.axhline(0, color='black')
plt.axvline(0, color='black')
plt.grid(True)
plt.title("Mapa de Navegação Vetorial")
plt.text(x, y, " Destino")
plt.text(ux, uy, " Unitário")
plt.text(0, limite-0.5, "N")
plt.text(0, -limite+0.5, "S")
plt.text(limite-0.5, 0, "L")
plt.text(-limite+0.5, 0, "O")
plt.show()

import math
import tkinter as tk

        if modulo == 0:

            return

        # Vetor unitário
        ux = x / modulo
        uy = y / modulo

        # Resultados
        resultado.config(
            text=f'Módulo: {modulo:.4f}\n'
                 f'Vetor unitário: ({ux:.4f}, {uy:.4f})'
        )

        # Gráfico
        plt.figure(figsize=(6, 6))
        plt.quiver(0, 0, x, y, angles='xy', scale_units='xy', scale=1, color='blue')
        plt.quiver(0, 0, ux, uy, angles='xy', scale_units='xy', scale=1, color='red')
        limite = max(abs(x), abs(y)) + 2
        plt.xlim(-limite, limite)
        plt.ylim(-limite, limite)
        plt.axhline(0, color='black')
        plt.axvline(0, color='black')
        plt.grid(True)
        plt.title("Visualização Geométrica dos Vetores")
        plt.text(x, y, " Vetor Original")
        plt.text(ux, uy, " Vetor Unitário")
        plt.show()

    except ValueError:
        messagebox.showerror("Erro", "Digite apenas números.")

# Interface
janela = tk.Tk()
janela.title("Visualizador Geométrico de Vetores")
janela.geometry("400x300")

tk.Label(janela, text="Visualização de Vetores", font="Arial 14").pack(pady=10)

tk.Label(janela, text="Digite x:").pack()
entrada_x = tk.Entry(janela)
entrada_x.pack()

tk.Label(janela, text="Digite y:").pack()
entrada_y = tk.Entry(janela)
entrada_y.pack()

tk.Button(janela, text="Calcular e Mostrar", command=calcular_e_desenhar).pack(pady=10)

resultado = tk.Label(janela, text="", font="Arial 12")
resultado.pack(pady=10)

janela.mainloop()
