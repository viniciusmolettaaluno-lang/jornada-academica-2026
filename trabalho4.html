# Exercício 1 — Conversão de Tempo
# Fonte: Claude.IA

def converter_tempo():
    # Entrada
    t = int(input("Digite o tempo em segundos: "))

    # Cálculos
    dias = t // 86400
    resto1 = t % 86400
    horas = resto1 // 3600
    resto2 = resto1 % 3600
    minutos = resto2 // 60
    segundos = resto2 % 60

    # Saída
    print(f"\n{dias} dias, {horas:02d}:{minutos:02d}:{segundos:02d}")

if __name__ == "__main__":
    converter_tempo()

# Exemplo de execução:

💻 Código Exemplo (Exercício 2 - Ponte de Wheatstone)

# Exercício 2 — Ponte de Wheatstone
# Fonte: ChatGPT

# Entrada de dados
R1 = float(input("Digite o valor de R1 (ohms): "))
R2 = float(input("Digite o valor de R2 (ohms): "))
R3 = float(input("Digite o valor de R3 (ohms): "))
Vfonte = float(input("Digite o valor da tensão da fonte (V): "))

print("\n--- Análise do Circuito ---")

# Validação
if R1 == 0:
    print("Erro: R1 não pode ser zero")
else:
    # Cálculos
    Rx = (R3 * R2) / R1
    ramo1 = R1 + Rx
    ramo2 = R2 + R3
    R_total = (ramo1 * ramo2) / (ramo1 + ramo2)
    I_total = Vfonte / R_total
    P_total = Vfonte * I_total

    # Saídas
    print(f"Resistência desconhecida (Rx): {Rx:.2f} ohms")
    print(f"Resistência total: {R_total:.2f} ohms")
    print(f"Corrente total: {I_total:.2f} A")
    print(f"Potência total: {P_total:.2f} W")

    # Verificação de equilíbrio
    if abs(R1 * Rx - R2 * R3) < 1e-6:
        print("A ponte está equilibrada")
    else:
        print("Status: ponte desequilibrada")

# Exercício 3 — Cálculo de Impostos (IPI, ICMS, PIS/COFINS)
# Fonte: Claude.IA

def calcular_nota_fiscal():
    # Entrada de dados
    preco_fabrica = float(input("Digite o preço da fábrica (R$): "))
    aliq_ipi = float(input("Digite a alíquota do IPI (%): "))
    aliq_icms = float(input("Digite a alíquota do ICMS (%): "))
    aliq_pis_cofins = float(input("Digite a alíquota PIS/COFINS (%): "))

    # Cálculos
    ipi = preco_fabrica * aliq_ipi / 100
    base_icms = preco_fabrica + ipi
    icms = base_icms * aliq_icms / 100
    base_pc = base_icms + icms
    pis_cofins = base_pc * aliq_pis_cofins / 100
    preco_final = base_pc + pis_cofins
    carga_total_perc = (preco_final - preco_fabrica) / preco_fabrica * 100

    # Demonstrativo
    print("\n=== Demonstrativo ===")
    print(f"Preço de fábrica: R$ {preco_fabrica:.2f}")
    print(f"Valor IPI: R$ {ipi:.2f}")
    print(f"Base ICMS (fábrica+IPI): R$ {base_icms:.2f}")
    print(f"Valor ICMS: R$ {icms:.2f}")
    print(f"Valor PIS/COFINS: R$ {pis_cofins:.2f}")
    print(f"Preço Final: R$ {preco_final:.2f}")
    print(f"Carga tributária total: {carga_total_perc:.2f}%")

if __name__ == "__main__":
    calcular_nota_fiscal()

# Exercício 4 — Classificação de Solo (SPT)
# Fonte: Claude.IA

# Entrada de dados
spt = int(input("Digite o valor do SPT: "))
cap_carga = float(input("Digite a capacidade de carga (kPa): "))

print("\n--- Classificação do Solo ---")

# Classificação principal
if spt < 5 or cap_carga < 80:
    print("INAPTO: reforço no solo necessário")
elif spt < 10 or cap_carga < 150:
    print("RESTRITO: fundação especial recomendada")
elif spt >= 30 and cap_carga >= 300:
    print("EXCELENTE: solo de alta capacidade")
else:
    print("ADEQUADO: fundação convencional")

# Informações adicionais
if spt < 10:
    print("SPT baixo")
if cap_carga < 150:
    print("Capacidade de carga baixa")
